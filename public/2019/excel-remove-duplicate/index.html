    Dim x
    'Updateby20140924
    With CreateObject("Scripting.Dictionary")
        .CompareMode = vbTextCompare
        For Each x In Split(txt, delim)

        Next
        If .Count > 0 Then RemoveDupes2 = Join(.keys, delim)
    End With
End Function
